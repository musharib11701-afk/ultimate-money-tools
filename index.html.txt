    <script>
        // 1. TOOL SWITCHING LOGIC
        function showTool(toolId) {
            document.querySelectorAll('.tool-card').forEach(card => card.classList.remove('active'));
            document.getElementById(toolId).classList.add('active');
        }

        // 2. METAMASK "MONEY" LOGIC
        async function connectMetaMask() {
            if (window.ethereum) {
                try {
                    const accounts = await window.ethereum.request({ method: 'eth_requestAccounts' });
                    document.getElementById('connectWallet').innerText = "Wallet: " + accounts[0].slice(0,6) + "...";
                    alert("Wallet Connected! You are ready to receive payments.");
                } catch (err) {
                    alert("Connection rejected.");
                }
            } else {
                alert("Please install MetaMask extension!");
            }
        }

        // 3. WORD UNSCRAMBLER LOGIC
        function unscramble() {
            let str = document.getElementById('scrambledInput').value.toLowerCase();
            if(!str || str.length > 8) {
                alert("Please enter 2-8 letters (longer words crash browsers!)");
                return;
            }
            const getPerms = (s) => {
                if (s.length <= 1) return [s];
                let p = [];
                for (let i = 0; i < s.length; i++) {
                    let char = s[i];
                    let rem = s.slice(0, i) + s.slice(i + 1);
                    for (let x of getPerms(rem)) p.push(char + x);
                }
                return [...new Set(p)];
            };
            document.getElementById('wordResult').innerText = "Combinations: " + getPerms(str).slice(0, 30).join(", ");
        }

        // 4. MATH LOGIC
        function calculateMath() {
            try {
                // Using eval with Caution - this allows advanced Math functions
                const result = eval(document.getElementById('mathInput').value);
                document.getElementById('mathResult').innerText = result;
            } catch {
                alert("Math error! Use format like: 5 * 5 or Math.sqrt(16)");
            }
        }

        // 5. SCIENCE LOGIC
        function solveForce() {
            let m = document.getElementById('mass').value;
            let a = document.getElementById('accel').value;
            document.getElementById('scienceResult').innerText = `Force = ${m * a} Newtons`;
        }
        function solveMoles() {
            let g = document.getElementById('grams').value;
            let mm = document.getElementById('molarMass').value;
            document.getElementById('scienceResult').innerText = `Moles = ${(g / mm).toFixed(4)} mol`;
        }

        // 6. LIVE CURRENCY LOGIC (Replace 'YOUR_KEY_HERE' when you have it)
        const API_KEY = 'YOUR_KEY_HERE'; 

        async function convertCurrency() {
            const amt = document.getElementById('amount').value;
            const from = document.getElementById('from').value;
            const to = document.getElementById('to').value;
            const res = document.getElementById('currResult');

            if (API_KEY === 'YOUR_KEY_HERE') {
                res.innerText = "Error: Set your API Key in the code!";
                return;
            }

            res.innerText = "Fetching...";
            try {
                const response = await fetch(`https://api.currencyfreaks.com/v2.0/rates/latest?apikey=${API_KEY}`);
                const data = await response.json();
                const result = (amt / data.rates[from]) * data.rates[to];
                res.innerText = `${amt} ${from} = ${result.toLocaleString(undefined, {maximumFractionDigits: 5})} ${to}`;
            } catch (e) {
                res.innerText = "Connection Failed.";
            }
        }
